// packages/ui/src/templates/hooks/use-entity-actions.ts

import { useCallback } from 'react';

export function useEntityActions<T extends Record<string, ActionDef>>(
  actions: T
) {
  const executeAction = useCallback(
    async <K extends keyof T>(action: K, input: T[K]['input']): Promise<T[K]['output']> => {
      const actionFn = actions[action];
      if (!actionFn) {
        throw new Error(`Action '${String(action)}' not found`);
      }
      return actionFn(input);
    },
    [actions]
  );

  const getAvailableActions = useCallback(() => {
    return Object.keys(actions) as (keyof T)[];
  }, [actions]);

  return {
    executeAction,
    getAvailableActions,
  };
}
